316  template <typename CharT, typename Traits>
317  static void print_val (std::basic_ostream<CharT, Traits> &os,
318  const std::string &val)
319  {
320  if (val.size() <= 60) {
321  os << val;
322  return;
323  }
324 
325  std::string::size_type pos = val.find(' ');
326  if (pos == std::string::npos) {
327  os << val;
328  return;
329  }
330 
331  std::string::size_type init = 0;
332  std::vector<std::string> val_tmp;
333  std::vector<std::string> val_vec;
334  while (pos != std::string::npos) {
335  val_tmp.push_back(val.substr(init, pos - init + 1));
336  init = pos + 1;
337  pos = val.find(' ', init);
338  }
339  for (std::size_t i = 0; i != val_tmp.size(); ++i) {
340  if (val_tmp[i] != std::string(" "))
341  val_vec.push_back(val_tmp[i]);
342  }
343 
344  if (val_vec.size() == 0)
345  return;
346 
347  if (val_vec.size() == 1) {
348  os << val_vec[0];
349  return;
350  }
351 
352  os << val_vec[0] << '\n';
353  for (std::size_t i = 1; i != val_vec.size() - 1; ++i)
354  os << std::setw(41) << ' ' << val_vec[i] << '\n';
355  os << std::setw(41) << ' ' << val_vec.back();
356  }
