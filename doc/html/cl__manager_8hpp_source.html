448  {

467  device_ = device_vec_[0];

485  bool platform_filter (cl_device_type dev_type)
486  {
487  std::vector<cl::Platform> platform_vec;
488  try {
489  cl::Platform::get(&platform_vec);
490  } catch (cl::Error) {
491  platform_vec.clear();
492  }
493  if (platform_vec.size() == 0)
494  return false;
495 
496  // If not using default platform
497  if (!setup_default_.default_platform()) {
498  for (std::size_t p = 0; p != platform_vec.size(); ++p) {
499  try {
500  std::string name;
501  platform_vec[p].getInfo(CL_PLATFORM_NAME, &name);
502  if (setup_default_.check_platform(name)) {
503  platform_ = platform_vec[p];
504  return true;
505  }
506  } catch (cl::Error) {}
507  }
508  }
509 
510  // Using default platform: finding the first that has the device type
511  for (std::size_t p = 0; p != platform_vec.size(); ++p) {
512  try {
513  std::vector<cl::Device> dev_pool;
514  std::vector<cl::Device> dev_select;
515  platform_vec[p].getDevices(dev_type, &dev_pool);
516  device_filter(dev_pool, dev_select);
517  if (dev_select.size() != 0) {
518  platform_ = platform_vec[p];
519  return true;
520  }
521  } catch (cl::Error) {}
522  }
523 
524  return false;
525  }
